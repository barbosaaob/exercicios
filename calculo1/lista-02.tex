\begin{enumerate}
    \vspace{0.5cm}
    \item Resolva as equa\c{c}\~oes (encontre apenas as solu\c{c}\~oes reais):
        \begin{enumerate}
            \vspace{0.3cm}
            \item $\ds x+5=14-\frac{1}{2}x$
            \vspace{0.3cm}
            \item $\ds\frac{2x}{x+1}=\frac{2x-1}{x}$
            \vspace{0.3cm}
            \item $\ds x^2-x-12=0$
            \vspace{0.3cm}
            \item $\ds 3|x-4|=10$
            \vspace{0.3cm}
            \item $2x{(4-x)}^{-1/2}-3\sqrt{4-x}=0$
        \end{enumerate}

    \vspace{0.5cm}
    \item Resolva as inequa\c{c}\~oes abaixo. Escreva a solu\c{c}\~ao usando a nota\c{c}\~ao de intervalo.
        \begin{enumerate}
            \vspace{0.3cm}
            \item $-4<5-3x\le17$
            \vspace{0.3cm}
            \item $x^2<2x+8$
            \vspace{0.3cm}
            \item $x(x-1)(x+2)>0$
            \vspace{0.3cm}
            \item $|x-4|<3$
            \vspace{0.3cm}
            \item $\ds\frac{2x-3}{x+1}\le 1$
        \end{enumerate}
\end{enumerate}

\vspace{1cm}
Respostas:
\begin{enumerate}
    \item
        \begin{enumerate*}
            \item $6$
            \item $1$
            \item $-3,4$
            \item $\frac{2}{3},\frac{22}{3}$
            \item $\frac{12}{5}$
        \end{enumerate*}
    \item
        \begin{enumerate*}
            \item $[-4,3)$
            \item $(-2,4)$
            \item $(-2,0) \cup (1,\infty)$
            \item $(1,7)$
            \item $(-1,4]$
        \end{enumerate*}
\end{enumerate}
