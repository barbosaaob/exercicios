\begin{enumerate}
	\vspace{0.5cm}
	\item Resolva:

		\begin{enumerate}
			\item $(3+2i)+(2-5i)$
			\item $(1+i)+(1-i)-2i$
			\item $(2-3i)(1+5i)$
			\item $(4-3i)(5-i)(1+i)$
			\item $(7+2i)(7-2i)$
			\item $(5+2i)(2-5i)-i$
		\end{enumerate}

	\vspace{0.5cm}
	\item Calcule as pot\^{e}ncias:
		
		\begin{enumerate}
			\item $i^{76}$
			\item $i^{110}$
			\item $i^{503}$
			\item $i^{2017}$
		\end{enumerate}

	\vspace{0.5cm}
	\item Prove que $(1-i)^2=-2i$ e calcule $z = (1-i)^{96}+(1-i)^{97}$.

	\vspace{0.5cm}
	\item Determine $x, y\in \mathbb{R}$ tais que 

		\begin{enumerate}
			\item $3+5ix = y-15i$
			\item $(x+yi)^2=2i$
			\item $(2-x+3y)+2yi=0$
			\item $(x+yi)(2+3i)=1+8i$
			\item $(3-i)(x+yi)=20$
		\end{enumerate}

	\vspace{0.5cm}
	\item Prove que $(a+bi)^2 = a^2 - b^2 + 2abi$.
\end{enumerate}
