            \item $9y''+12y'+4y=0$, $y(\pi)=5$, $y'(\pi)=-4$
            \item $9y''+12y'+4y=0$, $y(0)=1$, $y'(0)=0$
            \item $y''-6y'+10y=0$, $y(0)=2$, $y'(0)=3$
        \end{enumerate}
