%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%   formulario  inicio  %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{enumerate}
	\item Determine $a$, $b$, $c$ e $d$ tais que
		\[\begin{bmatrix}
			a-b & b+c \\
			3d+c & 2a-4d
		\end{bmatrix}
		=
		\begin{bmatrix}
			8 & 1 \\
			7 & 6
		\end{bmatrix}\]

	\vspace{0.5cm}
	\noindent{}Sejam
	\[A = \begin{bmatrix}
			3 & 0 \\
			-1 & 2 \\
			1 & 1
		\end{bmatrix},
	B = \begin{bmatrix}
			4 & -1 \\
			0 & 2
		\end{bmatrix},
	C = \begin{bmatrix}
			1 & 4 & 2 \\
			3 & 1 & 5
		\end{bmatrix},
	D = \begin{bmatrix}
			1 & 5 & 2 \\
			-1 & 0 & 1 \\
			3 & 2 & 4
		\end{bmatrix},
	E = \begin{bmatrix}
			6 & 1 & 3 \\
			-1 & 1 & 2 \\
			4 & 1 & 3
		\end{bmatrix}\]
	\item Usando as matrizes acima, resolva as opera\c{c}\~oes abaixo quando poss\'{\i}vel:
		\begin{enumerate}
			\item $D+E$
			\item $\tr(D)$
			\item $2A^T+C$
			\item $B-B^T$
			\item $\tr(C^TA^T+2E^T)$
			\item $B^T(CC^T-A^TA)$
			\item $(-AC)^T+5D^T$
		\end{enumerate}

	\vspace{0.5cm}
	\item Mostre que se $A$ e $B$ s\~ao matrizes $n\times{}n$, ent\~ao $\tr(A+B) = \tr(A) + \tr(B)$.
\end{enumerate}
